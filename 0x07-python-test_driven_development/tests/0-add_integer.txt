import unittest
from 0-add_integer.py import add_integer

class TestAddInteger(unittest.TestCase):
    def test_addition(self):
        self.assertEqual(add_integer(0, 1), 1)
        self.assertEqual(add_integer(0, 1.0), 1)
        self.assertEqual(add_integer(-10, -2), -12)
        self.assertEqual(add_integer(2), 100)

    def test_types(self):
        self.assertRaises(TypeError, add_integer, "one", 2)
        self.assertRaises(TypeError, add_integer, 1, "two")
        self.assertRaises(TypeError, add_integer, [1, 3], 2)
